set lin 200 pages 200
col OBJECT_NAME format a30
col owner format a30
col fc_size_g format 999,999.99
col t_size_g format 999,999.99
SELECT
    fc.objectnumber,
    db.owner,
    db.object_name,
    fc.fc_size_mb,
    db.t_size_mb
FROM
    (
        SELECT
            objectnumber,
            -- SUM(cachedsize) / 1024 / 1024 / 1024 fc_size_gb
            SUM(cachedsize) / 1024 / 1024 / 1024 fc_size_mb
        FROM
            fc_usage_et
        WHERE
            dbuniquename = 'DBM01'
        GROUP BY
            objectnumber
        HAVING
        --    SUM(cachedsize) / 1024 / 1024 / 1024 > 1   ----> above 1gb
            SUM(cachedsize) / 1024 / 1024  > 1   ----> 1m
    ) fc
    LEFT JOIN (
        SELECT
            co.object_name,
            co.data_object_id,
            co.owner,
            cm.t_size_mb
        FROM
            (
                SELECT
                    owner,
                    segment_name,
                --    SUM(bytes / 1024 / 1024 / 1024) t_size_gb
                    SUM(bytes / 1024 / 1024 ) t_size_mb
                FROM
                    cdb_segments
                WHERE
                    owner in( 'TPCH10T')
                GROUP BY
                    owner,segment_name
                HAVING
                --   SUM(bytes / 1024 / 1024 / 1024) > 1 ----> above 1gb
                  SUM(bytes / 1024 / 1024 ) > 1 ----> above 1g 
            ) cm,
            cdb_objects co
        WHERE
            cm.segment_name = co.object_name and cm.owner=co.owner
    ) db ON (
        TRIM(fc.objectnumber) = db.data_object_id
    )
ORDER BY fc.objectnumber
;

alter iormplan dbPlan=(
  (name=prod, flashcache=on, flashlog=on),

flashlog 性能主要影响 log file parallel write等待事件的响应事件（很少出现大于32ms情况，不会出现大于0.5s），而不是影响整个log file sync等待事件的响应时间。 

FL_ACTUAL_OUTLIERS : 0 表示没有出现日志文件写延时超过0.5秒 如果大于 1%的 FL_IO_W 指标，表示机械磁盘和PCI闪存IO非常糟糕。可能硬件问题或者配置不合理。
FL_DISK_IO_ERRS : 不为 0 表示日志文件所在机械磁盘存在质量问题
FL_FLASH_IO_ERRS : 不为 0 表示日志文件所在闪存存在质量问题
FL_IO_W_SKIP_BUSY : 不为 0 表示日志文件所在磁盘太慢，或者redo产生太多，以至于flashlog中有大量重做日志未及时写入在线日志文件中
FL_ACTUAL_OUTLIERS 判断磁盘是否太慢
FL_IO_W_SKIP_LARGE: 不为 0 表示512 flashlog太小，可以考虑增大flashlog大小 FL_IO_W_SKIP_NO_BUFFER这个指标值很大的话，也可以考虑增大flashlog大小
